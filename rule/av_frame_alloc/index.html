DataFlow::Node getSource(FunctionCall fc){

  and result.asExpr() = fc
}

  fc.getTarget().hasName("av_frame_remove_side_data")
  and result.asExpr() = fc.getArgument(0)
}

FunctionCall freeTarget(FunctionCall malloc){

}
